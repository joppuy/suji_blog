def my_geocoder(row):
    try:        #이 try와 except를 사용하응 이유 : 코드를 돌리는 과정에서 에러로 인한 멈춤을 없에고 해당되지 않는 것들은 넘기고 해당되는 것들만 추출하기 위함을 알아야한다.
        point = geolocator.geocode(row).point
        return pd.Series({'Latitude': point.latitude, 'Longitude': point.longitude})
    except:
        return None
